/etc/apache2/sites-available/hotelaurademallorca.conf

<VirtualHost *:443>
    ServerAdmin admin@hotelaurademallorca.com
    ServerName hotelaurademallorca.com
    ServerALias www.hotelaurademallorca.com
    DocumentRoot /var/www/html/hotelaurademallorca

    <Directory /var/www/html>
        AllowOverride All
        Require all granted
    </Directory>

    SSLEngine on
    SSLCertificateFile /etc/ssl/certs/hotelaurademallorca.crt
    SSLCertificateKeyFile /etc/ssl/private/hotelaurademallorca.key

    ErrorLog ${APACHE_LOG_DIR}/error.log
    CustomLog ${APACHE_LOG_DIR}/access.log combined
</VirtualHost>
